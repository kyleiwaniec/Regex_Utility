020        private DFAStatesList<DFAList<Integer>> dfaStatesList = new DFAStatesList<DFAList<Integer>>(); // extended ArrayList with "marked" data field
021        private DFAList<Integer> eClos = new DFAList<Integer>();
022        private DFAList<Integer> nextClos;
023        private DFAList<Integer> startStateList = new DFAList<Integer>();
024        private ArrayList<Integer> finalDFAStateList;
025        private Stack<int[]> dfaTransitionStack; // the int[] will hold 3 values fromState/input/toState

031        public DFA(NFA nfa, char[] language) throws GrumpyCatError{
032                
033                ListGraph l = nfa.getGraph();
034                int s = l.getStartState();
035                int langLength = language.length;
036                dfaTransitionStack = new Stack<int[]>();
037                finalDFAStateList = new ArrayList<Integer>();
038                startStateList = new DFAList<Integer>();
039                startStateList.mark(false);
040                startStateList.setLabel(dfaStatelabel);
041
042                // increment count of unmarked states in dfaStatesList:
043                dfaStatesList.updateUnMarkedStates(1);
044                startStateList.add(s); // startStateList now contains 1 state: the NFA startState.
045                
046                eClos = getEClosure(startStateList, l); // adds new closure to DFA states (dfaStatesList), and returns it.
047                eClos.mark(false);
048                eClos.setLabel(dfaStatelabel);
049                dfaStatesList.add(eClos);       
050                
051                // while there are unmarked states, do:
052                while(dfaStatesList.hasUnMarkedStates()){
053                        // loop over all the lists and return the first unmarked one.
054                        for(DFAList<Integer> m : dfaStatesList){
055                                //DFAList<Integer> mm = (DFAList<Integer>) m;
056                                if(!m.isMarked()){
057                                        eClos = m;
058                                        break;
059                                }
060                                
061                        }
062                        transition[0] = eClos.getLabel();
063                        eClos.mark(true);
064                        dfaStatesList.updateUnMarkedStates(-1);
065
066                        //mark this StateSet as final in DFA if it contains a final state from the NFA
067                        if(eClos.contains(nfa.nfaAcceptState())){
068                                if(!finalDFAStateList.contains(eClos.getLabel())){
069                                        finalDFAStateList.add(eClos.getLabel());
070                                }
071                        }
072                        
073
074                        nextClos = new DFAList<Integer>();
075                        
076                        for(int i = 0; i < langLength; i++){
077                                transition[1] = i; // the letter
078                                nextClos = getEClosure(move(eClos, i, language, l), l);
079                                int equalsCount = 0;
080                                for(DFAList<Integer> existingList : dfaStatesList){
081                                                //DFAList<Integer> existingList = (DFAList<Integer>) listObj;
082                                                if(existingList.equals(nextClos)){
083                                                        equalsCount++;
084                                        }
085                                }
086                                if(equalsCount == 0){
087                                        nextClos.setLabel(++dfaStatelabel);
088                                }
089
090                                // check if the list of DFA states alreaady contains the new closure
091                                // by iterating over all the lists and comparing
092                                if(equalsCount == 0 && nextClos.size() > 0){
093                                        nextClos.mark(false);
094                                        dfaStatesList.updateUnMarkedStates(1);
095                                        //nextClos.setLabel(++dfaStatelabel);
096                                        dfaStatesList.add(nextClos);
097                                }
098
099                                transition[2] = nextClos.getLabel();
100                                dfaTransitionStack.push(new int[] {transition[0], transition[1], transition[2]});
101                        }
102                        
103                }
104                int lbe = dfaStatelabel+1;
105                dfaTransTable = new int[lbe][langLength];
106                for(int i = 0; i < dfaTransTable.length; i++){
107                        for(int j = 0; j<langLength; j++){
108                                dfaTransTable[i][j] = -1;
109                        }
110                }
111                System.out.println("NFA to DFA State map: " +dfaStatesList);
112                
113                while(!dfaTransitionStack.empty()){
114                        StringBuilder sb = new StringBuilder();
115                        sb.append("[ ");
116                        int[] trans = dfaTransitionStack.pop();
117                                dfaTransTable[trans[0]][trans[1]] = trans[2];
118                                for(int j = 0; j<3; j++){
119                                        sb.append(trans[j]+" ");
120                                }
121                        sb.append("]"); 
122                        //System.out.println("dfaTransTable: "+sb);
123                }
124
125                System.out.println("DFA Transition Table (2D array of: DFA States x Alphabet)");
126                System.out.println("(-1 denotes unreachable states)");
127                printTable(dfaTransTable, language);
128
129        }
130        public DFAList<Integer> move(DFAList<Integer> S, int letter, char[] language, ListGraph l){
131                DFAList<Integer> list = new DFAList<Integer>();
132                for(Integer s : S){
133                        ArrayList<Edge> dll = l.get(s);
134                        for(Edge e : dll){
135                                if(e.getWeight() == language[letter]){
136                                        list.add(e.getDest());
137                                }
138                        }
139                }
140                return list;
141        }
142        public DFAList<Integer> getEClosure(DFAList<Integer> S, ListGraph l){ // StateSet S
143                Stack<Integer> stack = new Stack<Integer>();
144                DFAList<Integer> eClosure = new DFAList<Integer>();
145
146                for(Integer s : S){
147                        stack.push(s); // push state onto stack
148                        eClosure.add(s); // add state to eClosure list
149                }
150                // return all states reachable by epsilon transitions
151                while(!stack.empty()){
152                        int t = stack.pop();
153
154                        ArrayList<Edge> dll = l.get(t);
155                        for(Edge e : dll){
156                                if(e.isEpsilon()){
157                                        if(!eClosure.contains(e.getDest())){  /* destination state is not in the Closure */
158                                                stack.push(e.getDest());
159                                                eClosure.add(e.getDest());
160                                        }
161                                }
162                        }
163                }               
164                return eClosure;
165        }

172        public static void printTable(int[][] array, char[] language) {
173                System.out.print("  ");
174                for (int col = 0; col < language.length; col++) {
175                        System.out.printf("%-1s %-2c", " ", language[col]);
176                }
177                String separator = "----";
178                for(int ls = 0; ls<language.length; ls++){
179                        separator+="----";
180                }
181                System.out.println("\n   "+separator);
182                for (int j = 0; j < array.length; j++) {
183                        System.out.printf("%-2d", 0 +j);
184                        for (int i = 0; i < language.length; i++) {
185                                System.out.printf("%-1s %-2d", "|", array[j][i]);
186                        }
187                        System.out.println("|\n   "+separator);
188                }
189        }
