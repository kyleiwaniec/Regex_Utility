001package edu.mccc.cos210.regex;
002
003public interface RegexInt{
004
005        /** @return true if the regularExpression represented by this instance is found in the target beginning at 0 if no previous find or at the last end() position in the target.
006                throws NullPointerException if target is NULL.
007        */
008        boolean find(String target) throws NullPointerException;
009                
010        /** @return true if the regularExpression represented by this instance is found in the target beginning at position start
011        */      
012        boolean find(String target, int start) throws NullPointerException;
013
014        boolean match(String target);
015                // returns true if the regularExpression represented by this instance matches the target
016                // throws NullPointerException if target is NULL
017        int start(); 
018                // after the last successful find returns the offset into the target that the regular expression was found after a match returns 0
019                // throws IllegalStateException If no match has yet been attempted, or if the previous match operation failed.
020        int end(); 
021                // after the last successful find returns returns the offset after the last character found.
022                // after a match returns length of the target.
023                // throws IllegalStateException If no match has yet been attempted, or if the previous match operation failed.
024
025        String getCapture(int n); 
026                // after last successful match or find return the nth capture group. A value of 0       returns the entire found or matched  string.
027                // throws IllegalStateException If no match has yet been attempted, or if the previous match operation failed.
028                // throws IndexOutOfBoundsException If there is no capturing group in the pattern with the given index
029        int captureCount();
030                // after last successful match returns the number of strings captured in the target.
031                // throws IllegalStateException If no match has yet been attempted, or if the previous match operation failed.
032        void reset();
033                // resets the state of the Regex object so that it may be reused. 
034}
