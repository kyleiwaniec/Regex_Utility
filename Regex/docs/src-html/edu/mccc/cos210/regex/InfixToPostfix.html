024                Alternation |
025        */

032        /** First, insert concat operators into string 
033        */
034        public String insertConcatOps(String str) throws IOException{
035                BufferedReader br = stringToBR(str);
036                StringBuilder sb = new StringBuilder(); 
037                int c;
038                boolean isToken = false; // this is to keep track of tokens and only insert concat operators between tokens
039                boolean closingBracket = false; // this is to keep track of closing brackets and insert concat if the next token is NOT an operator.
040                boolean bracketExp = false;
041                boolean range = false;
042                boolean dashFlag = false;
043                boolean duplication = false;
044                int sbIdx = 0; // keeps track of number of charcters in converted regex sring
045
046                while ((c = br.read()) != -1){
047                        char curr = (char)c;
048                        if(bracketExp && !dashFlag && (char)c == '-'){ // the dash is not the first character after the bracket
049                                // process range
050                                // we need to know the PREVIOUS charcter and NEXT character to build range
051                                
052                                char prev = sb.charAt(sbIdx-1);
053                                int n = br.read(); //*********** CAREFUL - WE ARE ADVANCING THE READER ***********
054                                char next = (char)n;
055
056                                //INSERT range of characters including prev and next with | operators
057                                while(prev < next){
058                                        sb.append('|');
059                                        sbIdx++;
060                                        sb.append(++prev);
061                                        sbIdx++;
062                                }
063                                continue;               
064                        }
065                        if(bracketExp){
066                                dashFlag = false;
067                                // continue to process the dash as a regular character
068                        }
069                        if((char)c == '['){
070                                bracketExp = true;
071                                dashFlag = true;
072                        }
073                        if((char)c == ']'){
074                                bracketExp = false;
075                        }
076                        if(bracketExp){ // tokens inside bracket expressions use alternation, NOT concatenation.
077                                if(!isOperator((char)c) && isToken){
078                                        sb.append('|');
079                                        sbIdx++;
080                                }else if((char)c == '[' && (isToken || closingBracket)){
081                                        sb.append('•');
082                                        sbIdx++;
083                                }
084                                if(isOperator((char)c)){ 
085                                        isToken = false;
086                                }else{
087                                        isToken = true;
088                                }
089                        }else{
090                                if(!isOperator((char)c) && isToken){
091                                        sb.append('•');
092                                        sbIdx++;
093                                }else if((char)c == '(' && (isToken || closingBracket)){
094                                        sb.append('•');
095                                        sbIdx++;
096                                }
097                                if(isOperator((char)c)){
098                                        isToken = false;
099                                }else{
100                                        isToken = true;
101                                }
102                                if(closingBracket && isToken){
103                                        sb.append('•');
104                                        sbIdx++;
105                                }
106                                if(isToken && duplication){
107                                        sb.append('•');
108                                        sbIdx++;
109                                }
110                                
111                        }
112                        if(curr == '*' || curr == '+' || curr == '?'){
113                                duplication = true;
114                        }else{
115                                duplication = false;
116                        }
117                        if((char)c == ']' || (char)c == ')'){
118                                closingBracket = true;
119                        }else{
120                                closingBracket = false;
121                        }
122                        if((char)c == '['){
123                                sb.append('(');
124                        }else if((char)c == ']'){
125                                sb.append(')');
126                        }else{
127                                sb.append((char)c);
128                        }
129                        
130
131                        sbIdx++;
132                }
133                //System.out.println("with concat ops: "+sb.toString());
134                return sb.toString();
135        }

137        public String convert(String infix) throws IOException, GrumpyCatError{
138                operatorStack = new Stack<Character>();
139        postfix = new StringBuilder();
140        BufferedReader br = stringToBR(insertConcatOps(infix));
141        int c;
142        while ((c = br.read()) != -1){
143                char firstChar = (char)c;
144                if (isOperator(firstChar)) {
145                    processOperator(firstChar);
146                }else{
147                        postfix.append(firstChar);
148                        buildLanguage(firstChar);
149                }
150        }
151        // Pop any remaining operators and
152        // append them to postfix.
153        while (!operatorStack.empty()) {
154                char op = operatorStack.pop();
155                postfix.append(op);
156        }
157        return postfix.toString();
158        }

160        // private helper methods:
161        private static BufferedReader stringToBR(String str){
162                InputStream is = new ByteArrayInputStream(str.getBytes());
163                BufferedReader br = new BufferedReader(new InputStreamReader(is));
164                return br;
165        }

167        /**
168     * Method to process operators.
169     * @param op The operator
170     * @throws EmptyStackException
171     */
172    private void processOperator(char op) throws GrumpyCatError{
173        if (operatorStack.empty()) {
174            operatorStack.push(op);
175        } else {
176                //System.out.println("operatorStack: "+operatorStack);
177
178            // Peek the operator stack and
179            // let topOp be top operator.
180            char topOp = operatorStack.peek();
181            if(op == '('){
182                operatorStack.push(op);
183            }else if ((topOp == '(' || precedence(op) > precedence(topOp)) && op != ')') {
184                operatorStack.push(op);
185            }else if(op == ')'){
186                while(topOp != '('){
187                        operatorStack.pop();
188                        postfix.append(topOp);
189                        if (!operatorStack.empty()) {
190                        // Reset topOp.
191                        topOp = operatorStack.peek();
192                    }
193                }
194                operatorStack.pop(); // removes the right parenthesis
195                } else {
196                // Pop all stacked operators with equal
197                // or higher precedence than op.
198                while (!operatorStack.empty() && precedence(op) <= precedence(topOp) && topOp != '(') {
199                    operatorStack.pop();
200                    postfix.append(topOp);
201                    if (!operatorStack.empty()) {
202                        // Reset topOp.
203                        topOp = operatorStack.peek();
204                    }
205                }
206                // assert: Operator stack is empty or
207                //         current operator precedence >
208                //         top of stack operator precedence.
209                operatorStack.push(op);
210            }
211        }

213    public char[] getLanguage(){
214                int length = this.language.size();
215                char[] lang = new char[length];
216                for(int i = 0; i < length; i++){
217                        lang[i] = this.language.get(i);
218                }

222        public String languageToString(){
223                StringBuilder sb = new StringBuilder();
224                for(int i = 0; i < this.language.size(); i++) {
225                    sb.append(language.get(i));
226                }
227            return sb.toString();    
228        }
229        private void setLanguage(){
230                this.language = language;
231        }
232        private void buildLanguage(char c){
233                if(!this.language.contains(c)){
234                        this.language.add(c);
235                }
236        }
237        /**
238     * Determine whether a character is an operator.
239     * @param ch The character to be tested
240     * @return true if ch is an operator
241     */
242    private boolean isOperator(char ch) {
243        return OPERATORS.indexOf(ch) != -1;
244    }
245
246    /**
247     * Determine the precedence of an operator.
248     * @param op The operator
249     * @return the precedence
250     */
251    private int precedence(char op) {
252        return PRECEDENCE[OPERATORS.indexOf(op)];
253    }
