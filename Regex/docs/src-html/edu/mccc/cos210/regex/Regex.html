114                                break;
115                                case '$' : System.out.println("$"); 
116                                break;
117                                case '^' : System.out.println("^"); 
118                                break;
119                                default : 
120
121                                        // make a new edge
122                                        char[] ca = {(char)c};
123                                        listGraph.insert(new Edge(stackIdx, stackIdx+1, ca));

192        private static String fileToString(String pathToFile) {
193        String result = null;
194        DataInputStream in = null;
195        try {
196            File f = new File(pathToFile);
197            byte[] buffer = new byte[(int) f.length()];
198            in = new DataInputStream(new FileInputStream(f));
199            in.readFully(buffer);
200            result = new String(buffer, "UTF-8");
201        } catch (IOException e) {
202            throw new RuntimeException("IO problem in fileToString", e);
203        } finally {
204            try {
205                in.close();
206            } catch (IOException e) { 
207                    e.printStackTrace();
208            }
209        }
210        return result;
211    }
