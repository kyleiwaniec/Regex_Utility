087                BufferedReader br = stringToBR(regExpr);

234        private static String fileToString(String pathToFile) {
235        String result = null;
236        DataInputStream in = null;
237        try {
238            File f = new File(pathToFile);
239            byte[] buffer = new byte[(int) f.length()];
240            in = new DataInputStream(new FileInputStream(f));
241            in.readFully(buffer);
242            result = new String(buffer, "UTF-8");
243        } catch (IOException e) {
244            throw new RuntimeException("IO problem in fileToString", e);
245        } finally {
246            try {
247                in.close();
248            } catch (IOException e) { 
249                    e.printStackTrace();
250            }
251        }
252        return result;
