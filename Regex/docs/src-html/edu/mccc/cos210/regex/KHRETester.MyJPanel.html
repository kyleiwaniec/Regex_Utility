052
053                private static final long serialVersionUID = 1L;
054                private boolean result = false;
055                private JTextField jtfRegex = new JTextField("cat", 32);
056                private JTextField jtfString = new JTextField("12cat", 32);
057
058                private JButton jBtn_Match = new JButton("Match");
059                private JButton jBtn_Find = new JButton("Find Next");
060                private JButton jBtn_Reset = new JButton("Reset");
061                
062                private URL grumpy = getClass().getResource("/img/grumpycat.jpg");
063                private URL dancing = getClass().getResource("/img/dancingCat.gif");
064                private JLabel jLabel1 = new JLabel(new ImageIcon(grumpy));
065                private JLabel jLabel2 = new JLabel(new ImageIcon(dancing));
066                private int icon;
067
068                public MyJPanel() {
069                        super();
070                        jBtn_Match.addActionListener(this);
071                        jBtn_Find.addActionListener(this);
072                        jBtn_Reset.addActionListener(this);
073
074                        jLabel1.setVisible(false);
075                        jLabel2.setVisible(false);
076
077                        add(jtfRegex);
078                        add(jtfString);
079                        add(jBtn_Match);
080                        add(jBtn_Find);
081                        add(jBtn_Reset);
082                        add(jLabel1);
083                        add(jLabel2);
084                        
085                }
086
087                @Override
088                public void actionPerformed(ActionEvent e) {
089
090                        if (e.getSource() == jBtn_Match){
091                                try{
092                                        result = reEvaluator.match(jtfRegex.getText(), jtfString.getText());
093                                        icon = result ? 1 : 2;
094                                }catch (IOException ioe){
095                                        ioe.printStackTrace();
096                                }
097                        }else if (e.getSource() == jBtn_Find){
098                                try{
099                                        result = reEvaluator.find(jtfRegex.getText(), jtfString.getText(), reEvaluator.getStart());
100                                        icon = result ? 1 : 2;
101                                }catch (IOException ioe){
102                                        ioe.printStackTrace();
103                                }
104                        }else if (e.getSource() == jBtn_Reset){
105                                reEvaluator.reset();
106                                icon = 3;
107                        }
108
109                        System.out.println(
110                                "Evaluating '" +
111                                jtfRegex.getText() +
112                                "' using '" +
113                                jtfString.getText() +
114                                "'"
115                        );
116                        switch(icon){
117                                case 1 : jLabel1.setVisible(false);
118                                                 jLabel2.setVisible(true);
119                                break;
120                                case 2 : jLabel1.setVisible(true);
121                                                 jLabel2.setVisible(false);
122                                break;
123                                case 3 : jLabel1.setVisible(false);
124                                                 jLabel2.setVisible(false);                              
125                        }
126                        repaint();
127                }
