001package edu.mccc.cos210.tests;
002import edu.mccc.cos210.ds.PriorityQueue;
003import edu.mccc.cos210.ex.NoSuchElementException;
004
005public class PriorityQueueTest{
006        public static void main(String[] sa) throws NoSuchElementException{
007                String[] data = {"Francis", "Alice", "Betty", "Charlie", "Duke", "Gregory", "Thomas"};
008                PriorityQueue<String> pq = new PriorityQueue<String>();
009                for(int i = 0; i<data.length; i++){
010                        pq.offer(data[i]);
011                        //System.out.println("pqsize: "+pq.size());
012                }
013                System.out.println(pq.toString());
014                System.out.println("remove: " +pq.remove());
015                System.out.println(pq.toString());
016                System.out.println("peek:  "+pq.peek());
017                System.out.println(pq.toString());
018                System.out.println("poll:  "+pq.poll());
019                System.out.println(pq.toString());
020        }
021}
