014                        URL resource = getClass().getResource("/img/all.txt");
015
016                        File file = new File(resource.getPath());
017                        double filesize = (double) file.length()/1024/1024;
018                        System.out.println("the file is this big: "+filesize);
019                BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(file)));
020                String line = null;
021                int count = 0;
022
023                        long startTime = System.currentTimeMillis( );
024                        
025                HashTable<String, String> ht = new HashTable();
026                while( (line = br.readLine())!= null ){
027                        // \\s+ means any number of whitespaces between tokens
028                    String [] tokens = line.split("\\s+");
029
030                    for(int i = 0; i < tokens.length; i++){
031                      ht.put(tokens[i], tokens[i]);
032                      count++;
033                    }
034                }
035                long endTime = System.currentTimeMillis( );
036        
037                System.out.println( "Elapsed time: " + (endTime - startTime) );
038                System.out.println("number of words in the file: "+count);
039                System.out.println("number of keys: "+ht.getSize());
040                System.out.println("number of buckets: "+ht.getBuckeSize());
