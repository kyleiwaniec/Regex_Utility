052    /**
053     * Insert an item into the priority queue.
054     * @pre The ArrayList theData is in heap order.
055     * @post The item is in the priority queue and
056     *       theData is in heap order.
057     * @param item The item to be inserted
058     * @throws NullPointerException if the item to be inserted is null.

060    @Override
061    public boolean offer(E item) {
062        // Add the item to the heap.
063        theData.add(item);
064        System.out.println("added, now what"+theData.size());
065        // child is newly inserted item.
066        int child = theData.size() - 1;
067        int parent = (child - 1) / 2; // Find child's parent.
068        // Reheap
069        System.out.println("compare: "+compare(theData.get(parent), theData.get(child)));
070
071        while (parent >= 0 && compare(theData.get(parent), theData.get(child)) > 0) {
072            //System.out.println("before swap: "+theData.get(parent) +":"+theData.get(child));
073            swap(parent, child);
074            //System.out.println("after swap: "+theData.get(parent) +":"+theData.get(child));
075            child = parent;
076            parent = (child - 1) / 2;
077        }
078        return true;
079    }
080    public void swap(int a, int b){
081        E tempb = (E) theData.get(b);
082        E tempa = (E) theData.get(a);
083
084        theData.set(b, tempa);
085        theData.set(a, tempb);

096    /**
097     * Remove an item from the priority queue
098     * @pre The ArrayList theData is in heap order.
099     * @post Removed smallest item, theData is in heap order.
100     * @return The item with the smallest priority value or null if empty.
101     */
102    @Override
103    public E poll() {
104        if (isEmpty()) {
105            return null;
106        }
107        // Save the top of the heap. smallest item
108        E result = theData.get(0);
109        // If only one item then remove it.
110        if (theData.size() == 1) {
111            theData.remove(0);
112            return result;
113        }
114        // Remove the last item from the ArrayList and place it into
115        // the first position.
116        theData.set(0, theData.remove(theData.size() - 1));
117        // The parent starts at the top.
118        int parent = 0;
119        while (true) {
120            int leftChild = 2 * parent + 1;
121            if (leftChild >= theData.size()) {
122                break; // Out of heap.
123            }
124            int rightChild = leftChild + 1;
125            int minChild = leftChild; // Assume leftChild is smaller.
126            // See whether rightChild is smaller.
127            if (rightChild < theData.size() && compare(theData.get(leftChild), theData.get(rightChild)) > 0) {
128                minChild = rightChild;
129            }
130            // assert: minChild is the index of the smaller child.
131            // Move smaller child up heap if necessary.
132            if (compare(theData.get(parent), theData.get(minChild)) > 0) {
133                swap(parent, minChild);
134                parent = minChild;
135            } else { // Heap property is restored.
136                break;
137            }
138        }
139        return result;
140    }

145     /**
146     * Removes the entry at the front of the queue and returns it.
147     * @return The item removed if successful, throws exception if not
148     */
149    @Override
150    public E remove() throws NoSuchElementException{
151        if (theData.size() == 0) {
152            throw new NoSuchElementException();
153        } else {
154            E first = theData.get(0);
155            theData.remove(0);
156            return first;
157        }
158    }
159    /**
160     * Returns the item at the front of the queue without removing it.
161     * @return The item at the front if successful; null if not
162     */
163    @Override
164    public E peek() {
165        if (theData.size() == 0) {
166            return null;
167        } else {
168            return theData.get(0);
169        }

202    @Override
203    public String toString(){
204        String res = "";
205        for(int i = 0; i < theData.size(); i++) {
206            res += ", "+theData.get(i);
207        }
208        res = res.substring(2, res.length());

213    /**
214     * Compare two items using either a Comparator object�s compare method
215     * or their natural ordering using method compareTo.
216     * @pre If comparator is null, left and right implement Comparable<E>.
217     * @param left One item
218     * @param right The other item
219     * @return Negative int if left less than right,
220     *         0 if left equals right,
221     *         positive int if left > right
222     * @throws ClassCastException if items are not Comparable
223     */
224    @SuppressWarnings("unchecked")
225    private int compare(E left, E right) {
226        if (comparator != null) { // A Comparator is defined.
227            return comparator.compare(left, right);
228        } else { // Use left�s compareTo method.
229            return ((Comparable<E>) left).compareTo(right);
230        }
