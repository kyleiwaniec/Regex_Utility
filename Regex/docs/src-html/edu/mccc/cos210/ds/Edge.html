001package edu.mccc.cos210.ds;
002 
003 /** An Edge represents a relationship between two 
004 * vertices. 
005 */ 
006 public class Edge{ 
007         // Data Fields 
008         
009         /** The source vertex */ 
010         private int source; 
011         /** The destination vertex */ 
012         private int dest; 
013         /** The weight */ 
014         //private double weight; 
015
016         private char[] weight; 
017         
018         // Constructor 
019         /** Construct an Edge with a source of from 
020         * and a destination of to. Set the weight 
021         * to 1.0. 
022         * @param from - The source vertex 
023         * @param to - The destination vertex 
024         */ 
025         public Edge(int source, int dest) { 
026                 this.source = source; 
027                 this.dest = dest; 
028                 weight = new char[128];  // the ASCII set we're working with.
029         } 
030         
031         /** Construct a weighted edge with a source 
032         * of from and a destination of to. Set the 
033         * weight to w. 
034         * @param from - The source vertex 
035         * @param to - The destination vertex 
036         * @param w - The weight 
037         */ 
038         public Edge(int source, int dest, char[] w) { 
039                 this.source = source; 
040                 this.dest = dest; 
041                 this.weight = w; 
042         } 
043         
044         // Methods 
045         /** Get the source 
046         * @return The value of source 
047         */ 
048         public int getSource() { 
049                return source; 
050         } 
051         
052         /** Get the destination 
053         * @return The value of dest 
054         */ 
055         public int getDest() { 
056                return dest; 
057         } 
058         
059         /** Get the weight 
060         * @return the value of weight 
061         */ 
062         public char[] getWeight() { 
063                return weight; 
064         } 
065         
066         /** Return a String representation of the edge 
067         * @return A String representation of the edge 
068         */ 
069         @Override 
070         public String toString() { 
071                 StringBuilder sb = new StringBuilder(", [("); 
072                 sb.append(Integer.toString(source)); 
073                 sb.append(", "); 
074                 sb.append(Integer.toString(dest)); 
075                 sb.append("): "); 
076                 sb.append(String.valueOf(weight)); 
077                 sb.append("]"); 
078
079                
080
081                 return sb.toString(); 
082         } 
083         
084        
085         /** Return true if two edges are equal. Edges 
086         * are equal if the source and destination Programming Exercise Solutions Page 51 of 64
087         * are equal. Weight is not conidered.
088         * @param obj The object to compare to 
089         * @return true if the edges have the same source 
090         * and destination 
091         */ 
092         @Override 
093         public boolean equals(Object obj) { 
094                 if (obj instanceof Edge) { 
095                         Edge edge = (Edge) obj; 
096                         return (source == edge.source && dest == edge.dest); 
097                } else { 
098                        return false; 
099                } 
100         } 
101         
102         public boolean contains(char c){
103                for(int i = 0; i < weight.length; i++){
104                        if(c == weight[i]){
105                                return true;
106                        }
107                }
108                return false;
109         }
110
111         /** Return a hash code for an edge. The hash 
112         * code is the source shifted left 16 bits 
113         * exclusive or with the dest 
114         * @return a hash code for an edge 
115         */ 
116         @Override 
117         public int hashCode() { 
118                return (source << 16) ^ dest; 
119         } 
120}
