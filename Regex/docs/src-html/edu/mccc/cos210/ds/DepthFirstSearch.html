003import java.util.Iterator;
004
005/**
006 * Class to implement the depth-first search algorithm.
007 * @author Koffman and Wolfgang
008 **/
009public class DepthFirstSearch {
010
011    // Data Fields
012    /** A reference to the graph being searched. */
013    private Graph graph;
014    /** Array of parents in the depth-first search tree. */
015    private int[] parent;
016    /** Flag to indicate whether this vertex has been visited. */
017    private boolean[] visited;
018    /** The array that contains each vertex in discovery order. */
019    private int[] discoveryOrder;
020    /** The array that contains each vertex in finish order. */
021    private int[] finishOrder;
022    /** The index that indicates the discovery order. */
023    private int discoverIndex = 0;
024    /** The index that indicates the finish order. */
025    private int finishIndex = 0;
026
027    // Constructors
028    /**
029     * Construct the depth-first search of a Graph
030     * starting at vertex 0 and visiting the start vertices in
031     * ascending order.
032     * @param graph The graph
033     */
034    public DepthFirstSearch(Graph graph) {
035        this.graph = graph;
036        int n = graph.getNumV();
037        parent = new int[n];
038        visited = new boolean[n];
039        discoveryOrder = new int[n];
040        finishOrder = new int[n];
041        for (int i = 0; i < n; i++) {
042            parent[i] = -1;
043        }
044        for (int i = 0; i < n; i++) {
045            if (!visited[i]) {
046                depthFirstSearch(i);
047            }
048        }
049    }
050
051// Insert solution to programming exercise 1, section 4, chapter 10 here
052
053    /**
054     * Recursively depth-first search the graph
055     * starting at vertex current.
056     * @param current The start vertex
057     */
058    public void depthFirstSearch(int current) {
059        // Mark the current vertex visited.
060        visited[current] = true;
061        discoveryOrder[discoverIndex++] = current;
062        // Examine each vertex adjacent to the current vertex
063        Iterator<Edge> itr = graph.edgeIterator(current);
064        while (itr.hasNext()) {
065            int neighbor = itr.next().getDest();
066            // Process a neighbor that has not been visited
067            if (!visited[neighbor]) {
068                // Insert (current, neighbor) into the depth-
069                // first search tree.
070                parent[neighbor] = current;
071                // Recursively apply the algorithm
072                // starting at neighbor.
073                depthFirstSearch(neighbor);
074            }
075        }
076        // Mark current finished.
077        finishOrder[finishIndex++] = current;
078    }
079
080    /** Get the finish order 
081     * @return finish order 
082     */ 
083     public int[] getFinishOrder() { 
084        return finishOrder; 
085     } 
086     
087     /** Get the discovery order 
088     * @return discovery order 
089     */ 
090     public int[] getDiscoveryOrder() { 
091        return discoveryOrder; 
092     } 
093     
094     /** Get the parent 
095     * @return the parent 
096     */ 
097     public int[] getParent() { 
098        return parent; 
099     }
100
101}
