001package edu.mccc.cos210.ds;
002import java.util.Iterator;
003/** Itereface to specify the Graph ADT. 
004Each implementation should provide a constructor 
005that specifies the number of veritcies and whether 
006or not the grapgh is directed
007*/
008
009public interface Graph{
010        // Accessor methods
011        /** Return the number of veritces.
012                @return The number of vertices
013        */
014        int getNumV();
015
016        /** Determine whether the graph is directed.
017                @return true if the graph is directed graph
018        */
019        boolean isDirected();
020
021        /** Insert a new edge into the graph
022                @param edge The new Edge
023        */
024        void insert(Edge edge);
025        
026        /** Determine whether edge exists.
027                @param source The source vertex
028                @param dest The desitination vertex
029                @return true if there is an edge from source to dest.
030        */
031        boolean isEdge(int source, int dest);
032
033        /** Get the edge between two veritices.
034                @param source The source vertex
035                @param dest The desitination vertex
036                @return The Edge between these two vertices or 
037                an edge with a weight of Double.POSITIVE_INFINITY 
038                if there is no edge
039        */
040        Edge getEdge(int source, int dest);
041
042        /** Return an iterator to the edges connected to a given vertex
043                @param source The source vertex
044                @return An Iterator<Edge> to the vertices connected to the source
045
046        */
047        Iterator<Edge> edgeIterator(int source);
048        
049}
