003public class DoublyLinkedListImpl<E> implements DoublyLinkedListInt<E> {
004        private DNode<E> head;
005        private DNode<E> tail;
006        private DNode<E> current;
007        private int size = 0;
008        public DoublyLinkedListImpl() {
009                this.head = null;
010                this.tail = null;
011                this.current = null;
012        }
013        @Override
014        public E getFirst() throws java.util.NoSuchElementException {
015                if (this.size == 0) {
016                        throw new java.util.NoSuchElementException();
017                }
018                this.current = this.head;
019                return this.head.element;
020        }
021        @Override
022        public E getLast() throws java.util.NoSuchElementException {
023                if (this.size == 0) {
024                        throw new java.util.NoSuchElementException();
025                }
026                this.current = this.tail;
027                return this.tail.element;
028        }
029        @Override
030        public E getNext() throws java.util.NoSuchElementException {
031                if (this.size == 0 || this.current == null) {
032                        throw new java.util.NoSuchElementException();
033                }
034                this.current = this.current.next;
035                if (this.current == null) {
036                        throw new java.util.NoSuchElementException();
037                }
038                E element = this.current.element;
039                return element;
040        }
041        @Override
042        public E getPrevious() {
043                if (this.size == 0 || this.current == null) {
044                        throw new java.util.NoSuchElementException();
045                }
046                this.current = this.current.previous;
047
048                if (this.current == null) {
049                        throw new java.util.NoSuchElementException();
050                }
051                E element = this.current.element;
052                return element;
053        }
054        @Override
055        public E getCurrent() throws java.util.NoSuchElementException {
056                if (this.size == 0 || this.current == null) {
057                        throw new java.util.NoSuchElementException();
058                }
059                E element = this.current.element;
060                return element;
061        }
062        @Override 
063        public void addFirst(E element) {
064                DNode<E> node = new DNode<E>(element);
065                if(this.head != null){
066                        
067                        node.next = this.head;
068                        node.previous = null;
069
070                        this.head.previous = node;
071                        this.head = node;
072
073                }else{
074                        node.next = null;
075                        node.previous = null;
076                        this.tail = node;
077                        this.head = node;
078                }
079                this.current = node;
080                this.size++;
081
082
083        }
084        @Override 
085        public void addLast(E element) {
086                if (this.size == 0) { // or, check: this.tail == null
087                        addFirst(element);
088                } else {
089                        DNode<E> node = new DNode<E>(element);
090
091                        node.previous = this.tail;
092                        node.next = null;
093
094                        this.tail.next = node;
095                        this.tail = node;
096                        this.current = node;
097                        this.size++;
098                }
099
100        }
101        @Override 
102        public void addNext(E element) {
103                if (this.size == 0) {
104                        addFirst(element);
105                }else if(this.current == this.tail){
106                        addLast(element);
107                }else{
108                        DNode<E> node = new DNode<E>(element);
109
110                        node.previous = this.current;
111                        node.next = this.current.next;
112                        
113                        this.current.next.previous = node;
114                        this.current.next = node;
115
116                        this.current = node;
117                        this.size++;
118                }
119        }
120        @Override
121        public void addPrevious(E element) {
122                
123                if(this.current == this.head){
124                        throw new java.util.NoSuchElementException();
125                }else{
126                        DNode<E> node = new DNode<E>(element);
127                        node.next = this.current;
128                        node.previous = this.current.previous;
129
130
131                        this.current.previous.next = node;
132                        this.current.previous = node;
133                        this.current = node;
134                        this.size++;
135                }
136
137        }
138        @Override 
139        public void removeFirst() throws java.util.NoSuchElementException {
140                if (this.size == 0) {
141                        throw new java.util.NoSuchElementException();
142                }
143                
144
145                if(this.size == 1){
146                        this.current = null;
147                        this.head = null;
148                }else{
149                        this.current = this.head.next;
150                        this.head.next.previous = null;
151                }
152                
153                DNode<E> node = this.head;
154                this.head = node.next;
155                node.reset();
156
157                this.size--;
158
159                if(this.size == 1){
160                        this.tail = this.current;
161                }
162                
163        } 
164        @Override
165        public void removeLast() throws java.util.NoSuchElementException {
166                if (this.size == 0) {
167                        throw new java.util.NoSuchElementException();
168                }
169                
170                if (size == 1) {
171                        this.current = null;
172                        this.head = null;
173                } else {
174                        this.current = this.tail.previous;
175                        this.tail.previous.next = null;
176                }
177
178                DNode<E> node = this.tail;
179                this.tail = node.previous;
180
181                node.reset();
182                
183                this.size--;
184
185        }
186        @Override
187        public void removeNext() throws java.util.NoSuchElementException {
188                if (this.size == 0 || this.current == null || this.current.next == null) {
189                        this.current = null;
190                        throw new java.util.NoSuchElementException();
191                }
192                DNode<E> node = this.current.next;
193                this.current.next = node.next;
194                if (this.current.next != null) {
195                        this.current.next.previous = this.current;
196                } else {
197                        this.tail = this.current;
198                        //this.current = null;
199                }
200                node.reset();
201                this.size--;
202        }
203        @Override
204        public void removePrevious() throws java.util.NoSuchElementException {
205                if (this.size == 0 || this.current == null || this.current.previous == null) {
206                        this.current = null;
207                        throw new java.util.NoSuchElementException();
208                }
209                DNode<E> node = this.current.previous;
210                this.current.previous = node.previous;
211                if (this.current.previous != null) {
212                        this.current.previous.next = this.current;
213                } else {
214                        this.head = this.current;
215                        //this.current = null;
216                }
217                node.reset();
218                this.size--;
219        }
220
221        @Override
222        public int getSize() {
223                return this.size;
224        }
225        @Override
226        public boolean isEmpty() {
227                return this.size == 0;
228        }
229        @Override
230        public String toString() {
231                StringBuilder sb = new StringBuilder("< ");
232                DNode<E> node = head;
233                while (node != null) {
234                        if (node.element != null) {
235                                sb.append(node.element.toString() + " ");
236                        } else {
237                                sb.append("null ");
238                        }
239                        node = node.next;
240                }
241                sb.append(">");
242                return sb.toString();
243        }
244        private class DNode<T> {
245                private T element;
246                private DNode<T> next = null;
247                private DNode<T> previous = null;
248                public DNode(T element) {
249                        this.element = element;
250                }
251                private void reset() {
252                        this.element = null;
253                        this.next = null;
254                        this.previous = null;
255                }
256        }
