001package edu.mccc.cos210.ds;
002
003import java.io.IOException;
004import java.util.Scanner;
005
006/**
007 * Abstract base class for graphs. A graph is a set
008 * of vertices and a set of edges. Vertices are
009 * represented by integers from 0 to n - 1. Edges
010 * are ordered pairs of vertices.
011 * @author Koffman and Wolfgang
012 */
013public abstract class AbstractGraph implements Graph {
014
015    // Data Fields
016    /** The number of vertices */
017    private int numV;
018    /** Flag to indicate whether this is a directed graph */
019    private boolean directed;
020
021    // Constructor
022    /**
023     * Construct a graph with the specified number of vertices
024     * and the directed flag. If the directed flag is true,
025     * this is a directed graph.
026     * @param numV The number of vertices
027     * @param directed The directed flag
028     */
029    public AbstractGraph(int numV, boolean directed) {
030        this.numV = numV;
031        this.directed = directed;
032    }
033
034    // Accessor Methods
035    /**
036     * Return the number of vertices.
037     * @return The number of vertices
038     */
039    @Override
040    public int getNumV() {
041        return numV;
042    }
043
044    /**
045     * Return whether this is a directed graph.
046     * @return true if this is a directed graph
047     */
048    @Override
049    public boolean isDirected() {
050        return directed;
051    }
052
053    // Other Methods
054    /**
055     * Load the edges of a graph from the data in an input file.
056     * The file should contain a series of lines, each line
057     * with two or three data values. The first is the source,
058     * the second is the destination, and the optional third
059     * is the weight.
060     * @param bR The buffered reader containing the data
061     * @throws IOException if an I/O error occurs
062     */
063    public void loadEdgesFromFile(Scanner scan) throws IOException {
064         String line; 
065         while (scan.hasNextLine()) { 
066             line = scan.nextLine(); 
067             String[] tokens = line.split("\\s+"); 
068             int source = Integer.parseInt(tokens[0]); 
069             int dest = Integer.parseInt(tokens[1]); 
070             double weight = 1.0; 
071             if (tokens.length == 3) { 
072                weight = Double.parseDouble(tokens[2]); 
073             } 
074             insert(new Edge(source, dest, weight)); 
075         } 
076    }
077
078    /**
079     * Factory method to create a graph and load the data from an input
080     * file. The first line of the input file should contain the number
081     * of vertices. The remaining lines should contain the edge data as
082     * described under loadEdgesFromFile.
083     * @param scan The Scanner connected to the data file
084     * @param isDirected true if this is a directed graph,
085     *                   false otherwise
086     * @param type The string "Matrix" if an adjacency matrix is to be
087     *             created, and the string "List" if an adjacency list
088     *             is to be created
089     * @throws IllegalArgumentException if type is neither "Matrix"
090     *                                  nor "List"
091     */
092    public static Graph createGraph(Scanner scan,
093            boolean isDirected,
094            String type) throws IOException {
095        int numV = scan.nextInt();
096        scan.nextLine();
097        AbstractGraph returnValue = null;
098        if (type.equalsIgnoreCase("Matrix")) {
099            returnValue = new MatrixGraph(numV, isDirected);
100        } else if (type.equalsIgnoreCase("List")) {
101            returnValue = new ListGraph(numV, isDirected);
102        } else {
103            throw new IllegalArgumentException();
104        }
105        returnValue.loadEdgesFromFile(scan);
106        return returnValue;
107    }
108}
